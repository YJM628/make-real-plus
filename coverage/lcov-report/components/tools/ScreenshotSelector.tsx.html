/**
 * ScreenshotSelector Component
 * Feature: ai-html-visual-editor
 * Requirements: 8.1, 8.2, 8.3, 8.4
 * 
 * Provides screenshot selection functionality for AI-driven modifications.
 * Allows users to drag and draw a rectangular selection area over rendered HTML,
 * captures a screenshot of the selected region, and identifies all HTML elements
 * within the selection bounds.
 */
 
import * as React from 'react';
import { useState, useRef, useCallback, useEffect } from 'react';
// @ts-ignore - html2canvas has default export
import html2canvas from 'html2canvas';
 
/**
 * Rectangle bounds for selection area
 */
interface SelectionBounds {
  x: number;
  y: number;
  width: number;
  height: number;
}
 
/**
 * Props for ScreenshotSelector component
 */
export interface ScreenshotSelectorProps {
  /** Target HTML element to overlay and capture */
  targetElement: HTMLElement;
  
  /** Callback when selection is complete with screenshot and identified elements */
  onComplete: (screenshot: Blob, elements: HTMLElement[], bounds: DOMRect) => void;
  
  /** Callback when selection is cancelled */
  onCancel: () => void;
}
 
/**
 * ScreenshotSelector Component
 * 
 * Implements a screenshot selection tool with the following features:
 * - Semi-transparent overlay mask over the target element
 * - Drag-to-draw rectangular selection area
 * - Screenshot capture using html2canvas
 * - Element identification using bounding box intersection detection
 * - Returns screenshot Blob and list of elements within selection
 */
export const ScreenshotSelector: React.FC<ScreenshotSelectorProps> = ({
  targetElement,
  onComplete,
  onCancel,
}) => {
  const [isDrawing, setIsDrawing] = useState(false);
  const [selection, setSelection] = useState<SelectionBounds | null>(null);
  const [startPoint, setStartPoint] = useState<{ x: number; y: number } | null>(null);
  const overlayRef = useRef<HTMLDivElement>(null);
 
  /**
   * Handle mouse down - start drawing selection
   */
  const handleMouseDown = useCallback((e: React.MouseEvent<HTMLDivElement>) => {
    if (!overlayRef.current) return;
 
    const rect = overlayRef.current.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
 
    setStartPoint({ x, y });
    setIsDrawing(true);
    setSelection({ x, y, width: 0, height: 0 });
  }, []);
 
  /**
   * Handle mouse move - update selection rectangle
   */
  const handleMouseMove = useCallback((e: React.MouseEvent<HTMLDivElement>) => {
    if (!isDrawing || !startPoint || !overlayRef.current) return;
 
    const rect = overlayRef.current.getBoundingClientRect();
    const currentX = e.clientX - rect.left;
    const currentY = e.clientY - rect.top;
 
    // Calculate selection bounds (handle negative width/height for reverse dragging)
    const x = Math.min(startPoint.x, currentX);
    const y = Math.min(startPoint.y, currentY);
    const width = Math.abs(currentX - startPoint.x);
    const height = Math.abs(currentY - startPoint.y);
 
    setSelection({ x, y, width, height });
  }, [isDrawing, startPoint]);
 
  /**
   * Handle mouse up - complete selection and capture screenshot
   */
  const handleMouseUp = useCallback(async () => {
    if (!isDrawing || !selection || !overlayRef.current) return;
 
    setIsDrawing(false);
 
    // Ignore very small selections (likely accidental clicks)
    if (selection.width < 10 || selection.height < 10) {
      setSelection(null);
      setStartPoint(null);
      return;
    }
 
    try {
      // Calculate absolute bounds relative to the target element
      const overlayRect = overlayRef.current.getBoundingClientRect();
      const targetRect = targetElement.getBoundingClientRect();
      
      const absoluteBounds = new DOMRect(
        targetRect.left + selection.x,
        targetRect.top + selection.y,
        selection.width,
        selection.height
      );
 
      // Identify all elements within the selection bounds
      const elementsInBounds = identifyElementsInBounds(targetElement, absoluteBounds);
 
      // Capture screenshot of the selected area
      const screenshot = await captureScreenshot(targetElement, selection);
 
      // Call completion callback with results
      onComplete(screenshot, elementsInBounds, absoluteBounds);
    } catch (error) {
      console.error('Failed to capture screenshot:', error);
      // Reset state on error
      setSelection(null);
      setStartPoint(null);
    }
  }, [isDrawing, selection, targetElement, onComplete]);
 
  /**
   * Handle escape key to cancel selection
   */
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        onCancel();
      }
    };
 
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [onCancel]);
 
  return (
    <div
      ref={overlayRef}
      onMouseDown={handleMouseDown}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
      style={{
        position: 'absolute',
        top: 0,
        left: 0,
        width: '100%',
        height: '100%',
        backgroundColor: 'rgba(0, 0, 0, 0.3)',
        cursor: 'crosshair',
        zIndex: 9999,
      }}
    >
      {/* Selection rectangle */}
      {selection && (
        <div
          style={{
            position: 'absolute',
            left: `${selection.x}px`,
            top: `${selection.y}px`,
            width: `${selection.width}px`,
            height: `${selection.height}px`,
            border: '2px solid #007AFF',
            backgroundColor: 'rgba(0, 122, 255, 0.1)',
            pointerEvents: 'none',
          }}
        >
          {/* Selection dimensions display */}
          <div
            style={{
              position: 'absolute',
              top: '-24px',
              left: '0',
              padding: '2px 6px',
              backgroundColor: '#007AFF',
              color: 'white',
              fontSize: '12px',
              borderRadius: '3px',
              whiteSpace: 'nowrap',
            }}
          >
            {Math.round(selection.width)} Ã— {Math.round(selection.height)}
          </div>
        </div>
      )}
 
      {/* Cancel button */}
      <button
        onClick={onCancel}
        style={{
          position: 'absolute',
          top: '16px',
          right: '16px',
          padding: '8px 16px',
          backgroundColor: '#FF3B30',
          color: 'white',
          border: 'none',
          borderRadius: '6px',
          cursor: 'pointer',
          fontSize: '14px',
          fontWeight: '500',
          zIndex: 10000,
        }}
      >
        Cancel (Esc)
      </button>
    </div>
  );
};
 
/**
 * Capture screenshot of a specific area within an element
 * Requirements: 8.3
 * 
 * @param element - Target HTML element to capture
 * @param bounds - Selection bounds relative to the element
 * @returns Promise resolving to screenshot Blob
 */
async function captureScreenshot(
  element: HTMLElement,
  bounds: SelectionBounds
): Promise<Blob> {
  // Use html2canvas to capture the entire element
  const canvas = await html2canvas(element, {
    allowTaint: true,
    useCORS: true,
    logging: false,
    backgroundColor: null,
  });
 
  // Create a new canvas for the cropped region
  const croppedCanvas = document.createElement('canvas');
  croppedCanvas.width = bounds.width;
  croppedCanvas.height = bounds.height;
  
  const ctx = croppedCanvas.getContext('2d');
  if (!ctx) {
    throw new Error('Failed to get canvas context');
  }
 
  // Draw the cropped region from the full canvas
  ctx.drawImage(
    canvas,
    bounds.x, bounds.y, bounds.width, bounds.height,
    0, 0, bounds.width, bounds.height
  );
 
  // Convert canvas to Blob
  return new Promise((resolve, reject) => {
    // Check if toBlob is available (might not be in test environments)
    if (typeof croppedCanvas.toBlob === 'function') {
      croppedCanvas.toBlob((blob) => {
        if (blob) {
          resolve(blob);
        } else {
          reject(new Error('Failed to create blob from canvas'));
        }
      }, 'image/png');
    } else {
      // Fallback for environments without toBlob support (e.g., jsdom)
      try {
        const dataUrl = croppedCanvas.toDataURL('image/png');
        const base64 = dataUrl.split(',')[1];
        const binary = atob(base64);
        const array = new Uint8Array(binary.length);
        for (let i = 0; i < binary.length; i++) {
          array[i] = binary.charCodeAt(i);
        }
        const blob = new Blob([array], { type: 'image/png' });
        resolve(blob);
      } catch (error) {
        reject(error);
      }
    }
  });
}
 
/**
 * Identify all HTML elements within selection bounds using bounding box intersection
 * Requirements: 8.4
 * 
 * Uses intersection detection to find all elements whose bounding boxes
 * intersect with the selection rectangle.
 * 
 * @param root - Root element to search within
 * @param selectionBounds - Selection rectangle bounds
 * @returns Array of HTML elements within the selection
 */
function identifyElementsInBounds(
  root: HTMLElement,
  selectionBounds: DOMRect
): HTMLElement[] {
  const elementsInBounds: HTMLElement[] = [];
 
  /**
   * Check if two rectangles intersect
   */
  function rectsIntersect(rect1: DOMRect, rect2: DOMRect): boolean {
    return !(
      rect1.right < rect2.left ||
      rect1.left > rect2.right ||
      rect1.bottom < rect2.top ||
      rect1.top > rect2.bottom
    );
  }
 
  /**
   * Recursively traverse DOM tree and check for intersection
   */
  function traverse(element: HTMLElement) {
    // Get element's bounding box
    const elementBounds = element.getBoundingClientRect();
 
    // Check if element intersects with selection
    if (rectsIntersect(elementBounds, selectionBounds)) {
      // Only include elements with non-zero dimensions
      if (elementBounds.width > 0 && elementBounds.height > 0) {
        elementsInBounds.push(element);
      }
    }
 
    // Traverse children
    const children = element.children;
    for (let i = 0; i < children.length; i++) {
      const child = children[i];
      if (child instanceof HTMLElement) {
        traverse(child);
      }
    }
  }
 
  // Start traversal from root
  traverse(root);
 
  return elementsInBounds;
}
 
export default ScreenshotSelector;
